\begin{equation}
\label{eq:lagrangian}
\begin{split}
  \mathcal{L}&=\tfrac{1}{2}\Psibar (i \dslash-m) \Psi
  +\tfrac{1}{2}(\partial_\mu A\, \partial^\mu\! A-m^2 A^2)
  +\tfrac{1}{2}(\partial_\mu B\, \partial^\mu B-m^2 B^2)\\
  &-gmA(A^2+B^2)-\tfrac{1}{2}g^2(A^2+B^2)^2 
  -g\Psibar(A-i\gamma^5 B)\Psi.
\end{split}
\end{equation}
