\begin{equation}	\label{12.3}
	\begin{split}
\tope{T}^{\mu\nu}(x)
&=
\tope[0]{T}^{\mu\nu}(x) + \tope[\prime]{T}^{\mu\nu}(x)
\\
\tope[a]{T}^{\mu\nu}(x)
&=
\tope[a]{\pi}^{i\mu}(x) \circ \pd^\nu\tope{\varphi}_i(x)
		- \eta^{\mu\nu} \tope[a\mspace{-3mu}]{L}(x)
\quad
a=0,\prime
	\end{split}
	\end{equation}
