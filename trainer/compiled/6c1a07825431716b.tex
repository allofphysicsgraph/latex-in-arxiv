\begin{equation}               \label{eq:G}
\displaystyle{
\mathcal{G}_{K,L} \left( A, B; \omega^m \right) \equiv
\frac{1}{\sqrt{(K+L)!K!L!}} \partial_x^K  \partial_y^L
  \prod_{s=0}^{K+L-1} \left( \omega^{-s/2} x \mathbf{A} + 
        \omega^{s/2} y \mathbf{B} \right) \vert_{x=y=0}.  
}
\end{equation}
