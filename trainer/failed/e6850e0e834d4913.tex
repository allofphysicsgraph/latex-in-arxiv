\begin{split}
 E_\chi(A_+,A_-,\Phi;\theta)=&|\text{Pf}(2\pi\theta)|
 \Tr{}\left\{2|F_{A_+}|^2+2|F_{A_-}|^2+

 &\left.\qquad+\,\,4|\chi^+_+-\sigma^+(\Phi\otimes\Phi^\dagger)_0|^2+

\end{split}
