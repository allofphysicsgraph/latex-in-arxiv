\begin{equation}
\label{sunflux}
\begin{aligned}
   *H &= -\ex^{2\Phi} \diff (\ex^{-2\Phi}) 
      && \qquad \text{for $\SU(2)$} , \\ 
   *H &= -\ex^{2\Phi} \diff ( \ex^{-2\Phi} J ) 
      && \qquad \text{for $\SU(3)$} , \\
   *H &= -\ex^{2\Phi}\diff ( \ex^{-2\Phi}\tfrac{1}{2}J\wed J )
      && \qquad \text{for $\SU(4)$} .
\end{aligned}
\end{equation}
