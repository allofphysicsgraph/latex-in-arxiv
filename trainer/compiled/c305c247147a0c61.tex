    \int d^Dp\,\,\,
        p_{\lambda}p_{\tau}p_{\kappa}p_{\eta}\,\,\,
        e^{-\frac{1}{2}p\cdot R\cdot p}
    =
    (2\pi)^{D/2}
    \left(
        R^{-1}_{{\tau}{\lambda}}
            R^{-1}_{{\kappa}{\eta}}
        +R^{-1}_{{\kappa}{\lambda}}
            R^{-1}_{{\tau}{\eta}}
        +R^{-1}_{{\eta}{\lambda}}
            R^{-1}_{{\tau}{\kappa}}
    \right)
        (\det R)^{-1/2}.
