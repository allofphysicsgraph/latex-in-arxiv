\begin{equation}
\sum_{n=0}^\infty
\left\{
\frac{1}{(n+\frac{1}{2})^3}
\sum_{i_1=0}^n
\frac{1}{n!}\ \frac{(\frac{1}{2})_{(n+1)}}{(\frac{1}{2}+i_1)}
\right\}
-\sum_{n=1}^\infty
\left\{
\frac{1}{(n+\frac{1}{2})^2}
\underbrace{
\sum_{j_1=0}^n
\sum_{j_2=0}^{j_1-1}
}_{j_1>j_2}
\frac{1}{n!}\
\frac{(\frac{1}{2})_{(n+1)}}{(\frac{1}{2}+j_1)(\frac{1}{2}+j_2)}
\right\} = 0\ .
\label{12}
\end{equation}
