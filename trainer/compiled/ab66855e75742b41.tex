\begin{equation}
 \begin{array}{l}
[M_{rs}, M_{u v}]  =
 g_{rv} M_{su} -
 g_{ru} M_{sv} + g_{su} M_{rv} - g_{sv}M_{ru}\, ,
\cr
 [M_{rs}, P_{u} ]  =
g_{su}P_r-g_{ru}P_s\, ,
 \cr
[M_{rs}, K_{u} ]  =  g_{ru}K_s-g_{su}K_r\, ,
\cr
[M_{rs} , D] = 0 \, ,
\cr
[D, P_r ]  = - P_r\, , \qquad  \qquad [P_r, P_s ]  = 0 \, ,
\cr
[D, K_r ] =  K_r\, ,  \qquad \qquad [K_r, K_s ] = 0 \, ,
\cr
[K_r, P_s ] =   g_{rs} D + M_{rs} \, .
 \end{array}
\end{equation}
