\begin{equation}\label{eq:nonlocal constants}
 \begin{array}{rcl}
    H^{(-1)}(\lambda)[\psi] & = & \int_{x_-}^{x_+}dx\,\psi(\lambda)^2\,, \\
    H^{(-2)}(\lambda)[\psi] & =& \int_{x_-}^{x_+}dx\,\psi(\lambda)\,\bar{\psi}(\lambda)\,, \\
    H^{(-3)}(\lambda)[\psi] & =& \int_{x_-}^{x_+}dx\,\bar{\psi}(\lambda)^2\,. \
  \end{array}
\end{equation}
