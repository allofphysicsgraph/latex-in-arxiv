\begin{equation}
  \label{eq:a0period}
a_0(x_0+\beta,\vec x) = -T \log P \exp 
\left[-\int\limits_{x_0+\beta}^{x_0+2\beta} dt \, A_0(t,\vec x) \right] 
= -T \log P \exp 
\left[-\int\limits_{x_0}^{x_0+\beta} dt \, 
\underbrace{A_0(t+\beta,\vec x)}_{=A_0(t,\vec x)} \right] 
= a_0(x_0,\vec x)   \,.
\end{equation}
