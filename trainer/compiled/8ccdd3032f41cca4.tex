\begin{equation}
            \label{eq:CanPoisson}
            \{f,g\} = \sum_{i=1}^n \left( 
                \frac{\partial f}{\partial q^i}
                \frac{\partial g}{\partial p_i}
                -
                \frac{\partial f}{\partial p_i}
                \frac{\partial g}{\partial q^i}
            \right).
        \end{equation}
