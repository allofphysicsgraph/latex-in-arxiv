\begin{equation}
    \label{eq:starwexpl}
    \begin{split}
        f \starw g 
        &= N^{-1} (Nf \stars Ng) \\
        &=
        \sum_{r=0}^\infty \frac{1}{r!}
        \left(\frac{\hbar}{2\im}\right)^r
        \sum_{s=0}^r \binom{r}{s} (-1)^{r-s}
        \frac{\partial^r f}{\partial q^s \partial p^{r-s}}
        \frac{\partial^r g}{\partial q^{r-s} \partial p^s},
    \end{split}
\end{equation}
