  \begin{split}
    \th_3(u+v) \th_3(u-v) \th_3^2 &= \th_3(u)^2 \th_3(v)^2 +
    \th_1(u)^2 \th_1(v)^2  \\
    &= \th_4(u)^2 \th_4(v)^2 +\th_2(u)^2 \th_2(v)^2 \\
    \th_4(u+v) \th_4(u-v) \th_4^2 &= \th_3(u)^2 \th_3(v)^2 -
    \th_2(u)^2 \th_2(v)^2  \\
    &=\dots
  \end{split}
\end{equation}
