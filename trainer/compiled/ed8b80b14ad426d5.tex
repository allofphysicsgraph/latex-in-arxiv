\begin{equation}\label{deltaresult}
\delta(t)=
\begin{cases}
\delta(t_0)-\frac{\dot{\delta}(t_0)t_0}{1-3q}+\frac{\dot{\delta}(t_0)t_0^{3q}}{1-3q}t^{1-3q},
\hspace{1.4cm} q\neq \frac{1}{3} \\
\delta(t_0)+\dot{\delta}(t_0)t_0\ln(\frac{t}{t_0}), \hspace{2.6cm}
q=\frac{1}{3}
\end{cases}
\end{equation}
