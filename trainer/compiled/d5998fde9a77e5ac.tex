\begin{equation}
\label{eq:Lorentz-algebra}
    [J^K_I, J^M_N] = i \left(
                     \delta^K_N J^M_I -
                     \delta^M_I J^K_N +
                     \eta_{N\!I}\,\eta^{M\!L} J^K_L -
                     \eta_{N\!L}\,\eta^{M\!K} J^L_I
                     \right) \ .
\end{equation}
