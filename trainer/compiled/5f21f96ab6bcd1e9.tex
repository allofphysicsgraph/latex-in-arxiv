\begin{equation}
    \label{eq:JacobiLocal}
    \sum_{\ell} \left(
        \alpha^{i\ell} \frac{\partial \alpha^{jk}}{\partial x^\ell}
        + \alpha^{j\ell} \frac{\partial \alpha^{ki}}{\partial x^\ell}
        + \alpha^{k\ell} \frac{\partial \alpha^{ij}}{\partial x^\ell}
    \right) = 0.
\end{equation}
