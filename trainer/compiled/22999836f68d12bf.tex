\label{eq:massen}
T\ll m:\, \left\{\,
\begin{array}{rl}
  m_A^2&=m^2-\tfrac{20}{3}g^2\alpha m^2\\[1mm]

\end{array}\right.

\begin{array}{rl}
  m_A^2&=m^2-2g^2 T^2\\[1mm]
  m_B^2&=m^2\\[1mm]
  m_\Psi^2&=m^2-g^2 T^2
\end{array}\right.
