\label{eq:gR}
\begin{split}
  g \mathcal{R}(G,B)^{-1} &= 
  \begin{twomatrix}
    a + b E & a - b E^t \\
    c + d E & c - d E^t
  \end{twomatrix}
  = \begin{twomatrix}
      \Id_m &  \Id_m \\
        E'  & -{E'}^t
    \end{twomatrix}
    \begin{twomatrix}
      a + b E &     0 \\
         0    & a - b E^t
    \end{twomatrix} \\
  &= \mathcal{R}(G',B')^{-1} M(g,E).
\end{split}
\end{equation}
